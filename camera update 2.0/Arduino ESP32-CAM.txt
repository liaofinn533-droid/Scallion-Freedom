#include <Arduino.h>
#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <HTTPClient.h>
#include "esp_camera.h"
#include "soc/soc.h"
#include "soc/rtc_cntl_reg.h"
#include <base64.h> 


// 1. NETWORK & ENDPOINT CONFIGURATION
const char* ssid = "";
const char* password = "";
const char* server_url = "https://scallion-freedom-function-356495094540.europe-west2.run.app";

#define TIME_TO_SLEEP  3600000000ULL 


// 2. HARDWARE PIN DEFINITIONS (AI-Thinker ESP32-CAM)
#define PWDN_GPIO_NUM     32
#define RESET_GPIO_NUM    -1
#define XCLK_GPIO_NUM     0
#define SIOD_GPIO_NUM     26
#define SIOC_GPIO_NUM     27
#define Y9_GPIO_NUM       35
#define Y8_GPIO_NUM       34
#define Y7_GPIO_NUM       39
#define Y6_GPIO_NUM       36
#define Y5_GPIO_NUM       21
#define Y4_GPIO_NUM       19
#define Y3_GPIO_NUM       18
#define Y2_GPIO_NUM       5
#define VSYNC_GPIO_NUM    25
#define HREF_GPIO_NUM     23
#define PCLK_GPIO_NUM     22


// 3. CORE FUNCTIONS


/**
 * Captures, encodes, and attempts to upload the image.
 * Returns: true if upload successful (HTTP 200), false otherwise.
 */
bool capture_and_upload() {
  Serial.println("[SYSTEM] Initiating image acquisition sequence...");
  
  camera_fb_t *fb = esp_camera_fb_get();
  if (!fb) {
    Serial.println("[ERROR] Camera capture failed. Hardware malfunction suspected.");
    return false; // Return failure
  }
  Serial.printf("[INFO] Image captured. Buffer size: %u bytes.\n", fb->len);

  HTTPClient http;
  WiFiClientSecure client;
  client.setInsecure(); 

  Serial.printf("[NETWORK] Connecting to Cloud Endpoint: %s\n", server_url);
  http.setFollowRedirects(HTTPC_FORCE_FOLLOW_REDIRECTS);
  http.setTimeout(60000); // 60s Timeout

  bool result = false; // Default to fail

  if (http.begin(client, server_url)) {
    Serial.println("[PROCESSING] Encoding binary data to Base64 string...");
    String imageFile = base64::encode(fb->buf, fb->len);
    
    Serial.println("[PROCESSING] Constructing JSON payload...");
    String payload = "{\"image\": \"" + imageFile + "\", \"filename\": \"esp32_onion_monitor.jpg\"}";
    
    http.addHeader("Content-Type", "application/json");
    
    Serial.println("[NETWORK] Transmitting payload... (This may take time)");
    int httpCode = http.POST(payload);

    if (httpCode == 200) { // Only accept 200 OK as success
      Serial.printf("[SUCCESS] Transmission complete. HTTP Status: %d\n", httpCode);
      String response = http.getString();
      Serial.println("[CLOUD RESPONSE] " + response);
      result = true; // Mark as success
    } else {
      Serial.printf("[ERROR] Transmission failed. HTTP Status: %d. Error: %s\n", httpCode, http.errorToString(httpCode).c_str());
      result = false;
    }
    http.end(); 
  } else {
    Serial.println("[ERROR] Unable to establish connection to server.");
    result = false;
  }
  
  esp_camera_fb_return(fb); // Important: Release memory
  return result;
}


// 4. MAIN EXECUTION FLOW
void setup() {
  WRITE_PERI_REG(RTC_CNTL_BROWN_OUT_REG, 0); 

  Serial.begin(115200);
  Serial.println("\n\n--- Scallion Freedom: Reliable Edge Unit Started ---");

  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;
  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;
  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;
  config.pin_sccb_sda = SIOD_GPIO_NUM;
  config.pin_sccb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;
  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_JPEG;

  if(psramFound()){
    config.frame_size = FRAMESIZE_HVGA; 
    config.jpeg_quality = 12; 
    config.fb_count = 1; 
  } else {
    config.frame_size = FRAMESIZE_CIF;
    config.jpeg_quality = 12;
    config.fb_count = 1;
  }

  esp_err_t err = esp_camera_init(&config);
  if (err != ESP_OK) {
    Serial.printf("[FATAL] Camera init failed 0x%x\n", err);
    return;
  }

  sensor_t * s = esp_camera_sensor_get();
  if (s != NULL) {
      Serial.println("[SYSTEM] Applying Image Correction Parameters...");
      s->set_whitebal(s, 1);       
      s->set_awb_gain(s, 1);       
      s->set_wb_mode(s, 0);        
      s->set_dcw(s, 1);            // Remove green tint

      s->set_exposure_ctrl(s, 1);  
      s->set_aec2(s, 1);           
      s->set_ae_level(s, 2);       // Brighter
      s->set_gain_ctrl(s, 1);      
      s->set_gainceiling(s, (gainceiling_t)6); 
      
      s->set_brightness(s, 2);     
      s->set_contrast(s, 0);       
      s->set_saturation(s, 0);     
      s->set_raw_gma(s, 1);        // Gamma correction
      s->set_lenc(s, 1);           
  }


  WiFi.begin(ssid, password);
  Serial.print("[NETWORK] Connecting WiFi");
  int retries = 0;
  while (WiFi.status() != WL_CONNECTED && retries < 60) {
    delay(500);
    Serial.print(".");
    retries++;
  }
  
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println(" Connected.");
    
    bool uploadSuccess = false;
    int attempt = 1;

    while (uploadSuccess == false) {
      Serial.printf("\n[SYSTEM] Upload Attempt #%d\n", attempt);
      
 
      uploadSuccess = capture_and_upload();

      if (!uploadSuccess) {
        Serial.println("[WARNING] Upload failed. Retrying in 5 seconds...");
        delay(5000); // Wait 5 seconds before trying again
        
        if (WiFi.status() != WL_CONNECTED) {
           Serial.println("[NETWORK] WiFi lost. Reconnecting...");
           WiFi.reconnect();
           delay(5000);
        }
        attempt++;
      }
    }
    // If we break the loop, it means uploadSuccess is true!

  } else {
    Serial.println("\n[FATAL] WiFi connection failed.");
  }

  Serial.println("[SYSTEM] Task cycle complete. Entering Deep Sleep.");
  Serial.println("Goodnight Zzz..."); 
  Serial.flush(); 
  
  esp_sleep_enable_timer_wakeup(TIME_TO_SLEEP);
  esp_deep_sleep_start();
}

void loop() {
}
